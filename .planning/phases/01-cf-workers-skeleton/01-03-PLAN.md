// Uses cloudflare:test module (provided by @cloudflare/vitest-pool-workers)
import { env, createExecutionContext, waitOnExecutionContext } from 'cloudflare:test'
import worker from '../src/index'

export async function fetchWorker(path: string, init?: RequestInit) {
  const req = new Request(`http://localhost${path}`, init)
  const ctx = createExecutionContext()
  const res = await worker.fetch(req, env, ctx)
  await waitOnExecutionContext(ctx)
  return res
}

export { env }

Write worker/test/middleware/auth.test.ts:
```typescript
import { describe, it, expect } from 'vitest'
import { fetchWorker } from '../setup'

describe('JWT auth middleware', () => {
  it('returns 401 when Authorization header is missing', async () => {
    const res = await fetchWorker('/api/profile')
    expect(res.status).toBe(401)
  })

  it('returns 401 when JWT is malformed', async () => {
    const res = await fetchWorker('/api/profile', {
      headers: { Authorization: 'Bearer not.a.valid.jwt' },
    })
    expect(res.status).toBe(401)
  })

  it('returns 401 when JWT is expired', async () => {
    // Create an expired token: header.payload.signature with exp in the past
    // Base64url encode a payload with exp=1 (Unix epoch 1970)
    const header = btoa(JSON.stringify({ alg: 'HS256', typ: 'JWT' }))
      .replace(/\+/g, '-').replace(/\//g, '_').replace(/=/g, '')
    const payload = btoa(JSON.stringify({ sub: 1, openid: 'test', iat: 1, exp: 1 }))
      .replace(/\+/g, '-').replace(/\//g, '_').replace(/=/g, '')
    const fakeToken = `${header}.${payload}.fakesignature`
    const res = await fetchWorker('/api/profile', {
      headers: { Authorization: `Bearer ${fakeToken}` },
    })
    expect(res.status).toBe(401)
  })
})
```
