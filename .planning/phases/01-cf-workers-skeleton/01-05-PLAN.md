import { Hono } from 'hono'
import type { Bindings, Variables } from '../types'
import { ok, fail } from '../lib/response'
import { toCamel } from '../lib/db'
import { authMiddleware } from '../middleware/auth'

const router = new Hono<{ Bindings: Bindings; Variables: Variables }>()
router.use('*', authMiddleware)

router.get('/profile', async (c) => {
  const userId = c.get('userId')
  const row = await c.env.DB
    .prepare('SELECT * FROM profiles WHERE user_id = ?1 LIMIT 1')
    .bind(userId)
    .first<Record<string, unknown>>()
  if (!row) return fail(c, 404, 'PROFILE_NOT_FOUND', 'Profile not found')
  return ok(c, toCamel(row))
})

router.post('/profile', async (c) => {
  const userId = c.get('userId')
  const body = await c.req.json<{ name?: string; birthDate?: string; gender?: string }>()
    .catch(() => ({} as { name?: string; birthDate?: string; gender?: string }))

  if (!body.name || !body.birthDate || !body.gender) {
    return fail(c, 400, 'PROFILE_INVALID', 'name, birthDate, gender are required')
  }

  const existing = await c.env.DB
    .prepare('SELECT id FROM profiles WHERE user_id = ?1 LIMIT 1')
    .bind(userId)
    .first<{ id: number }>()

  if (existing) {
    await c.env.DB
      .prepare("UPDATE profiles SET name=?1, birth_date=?2, gender=?3, updated_at=datetime('now') WHERE user_id=?4")
      .bind(body.name, body.birthDate, body.gender, userId)
      .run()
    return ok(c, null)
  }

  await c.env.DB
    .prepare('INSERT INTO profiles (user_id, name, birth_date, gender) VALUES (?1,?2,?3,?4)')
    .bind(userId, body.name, body.birthDate, body.gender)
    .run()
  return ok(c, null, 201)
})

export default router

import { Hono } from 'hono'
import type { Bindings, Variables } from '../types'
import { ok, fail } from '../lib/response'
import { toCamel } from '../lib/db'
import { authMiddleware } from '../middleware/auth'

const router = new Hono<{ Bindings: Bindings; Variables: Variables }>()
router.use('*', authMiddleware)

router.get('/records', async (c) => {
  const userId = c.get('userId')
  const page = Math.max(1, parseInt(c.req.query('page') ?? '1'))
  const pageSize = Math.min(100, Math.max(1, parseInt(c.req.query('pageSize') ?? '20')))
  const offset = (page - 1) * pageSize

  const [rows, countRow] = await Promise.all([
    c.env.DB
      .prepare('SELECT * FROM records WHERE user_id=?1 AND deleted_at IS NULL ORDER BY created_at DESC LIMIT ?2 OFFSET ?3')
      .bind(userId, pageSize, offset)
      .all<Record<string, unknown>>(),
    c.env.DB
      .prepare('SELECT COUNT(*) as total FROM records WHERE user_id=?1 AND deleted_at IS NULL')
      .bind(userId)
      .first<{ total: number }>(),
  ])

  return c.json({
    success: true,
    data: (rows.results ?? []).map(toCamel),
    pagination: { page, pageSize, total: countRow?.total ?? 0 },
  })
})

router.post('/records', async (c) => {
  const userId = c.get('userId')
  const body = await c.req.json<{
    activityId?: string
    initType?: string
    focusSec?: number
    emotion?: string
    note?: string
  }>().catch(() => ({} as any))

  if (!body.activityId || !body.initType || body.focusSec === undefined || !body.emotion) {
    return fail(c, 400, 'RECORD_INVALID', 'activityId, initType, focusSec, emotion are required')
  }

  await c.env.DB
    .prepare('INSERT INTO records (user_id, activity_id, init_type, focus_sec, emotion, note) VALUES (?1,?2,?3,?4,?5,?6)')
    .bind(userId, body.activityId, body.initType, body.focusSec, body.emotion, body.note ?? null)
    .run()
  return ok(c, null, 201)
})

router.put('/records/:id', async (c) => {
  const userId = c.get('userId')
  const id = parseInt(c.req.param('id'))
  if (isNaN(id)) return fail(c, 400, 'RECORD_INVALID_ID', 'id must be a number')

  const row = await c.env.DB
    .prepare('SELECT id FROM records WHERE id=?1 AND user_id=?2 AND deleted_at IS NULL')
    .bind(id, userId)
    .first()
  if (!row) return fail(c, 404, 'RECORD_NOT_FOUND', 'Record not found')

  const body = await c.req.json<{ initType?: string; focusSec?: number; emotion?: string; note?: string }>()
    .catch(() => ({} as any))

  // Build SET clause dynamically for partial updates
  const updates: string[] = ["updated_at=datetime('now')"]
  const bindings: (string | number | null)[] = []
  let idx = 1

  if (body.initType !== undefined) { updates.push(`init_type=?${idx++}`); bindings.push(body.initType) }
  if (body.focusSec !== undefined) { updates.push(`focus_sec=?${idx++}`); bindings.push(body.focusSec) }
  if (body.emotion !== undefined) { updates.push(`emotion=?${idx++}`); bindings.push(body.emotion) }
  if (body.note !== undefined) { updates.push(`note=?${idx++}`); bindings.push(body.note) }

  if (bindings.length === 0) return ok(c, null) // nothing to update

  bindings.push(id) // for WHERE id=?N
  await c.env.DB
    .prepare(`UPDATE records SET ${updates.join(', ')} WHERE id=?${idx}`)
    .bind(...bindings)
    .run()
  return ok(c, null)
})

router.delete('/records/:id', async (c) => {
  const userId = c.get('userId')
  const id = parseInt(c.req.param('id'))
  if (isNaN(id)) return fail(c, 400, 'RECORD_INVALID_ID', 'id must be a number')

  const row = await c.env.DB
    .prepare('SELECT id FROM records WHERE id=?1 AND user_id=?2 AND deleted_at IS NULL')
    .bind(id, userId)
    .first()
  if (!row) return fail(c, 404, 'RECORD_NOT_FOUND', 'Record not found')

  await c.env.DB
    .prepare("UPDATE records SET deleted_at=datetime('now'), updated_at=datetime('now') WHERE id=?1")
    .bind(id)
    .run()
  return ok(c, null)
})

export default router
