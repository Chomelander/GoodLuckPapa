Now implement worker/src/routes/auth.ts:
```typescript
import { Hono } from 'hono'
import { sign } from 'hono/jwt'
import type { Bindings, Variables } from '../types'
import { ok, fail } from '../lib/response'
import { upsertUser } from '../lib/db'

const router = new Hono<{ Bindings: Bindings; Variables: Variables }>()

router.post('/auth/login', async (c) => {
  const body = await c.req.json<{ code?: string }>().catch(() => ({} as { code?: string }))

  if (!body.code) {
    return fail(c, 400, 'AUTH_MISSING_CODE', 'code is required')
  }

  // Exchange code for openid via WeChat jscode2session API (D-19, D-20)
  const url = `https://api.weixin.qq.com/sns/jscode2session?appid=${c.env.WX_APP_ID}&secret=${c.env.WX_APP_SECRET}&js_code=${body.code}&grant_type=authorization_code`

  let wxData: { openid?: string; errcode?: number; errmsg?: string }
  try {
    const wxRes = await fetch(url)
    wxData = await wxRes.json<typeof wxData>()
  } catch (err) {
    console.error('[auth] WeChat API fetch failed:', err)
    return fail(c, 500, 'AUTH_WX_NETWORK', 'Failed to reach WeChat API')
  }

  if (!wxData.openid) {
    return fail(c, 400, 'AUTH_WX_FAILED', wxData.errmsg ?? 'WeChat login failed')
  }

  // Upsert user in D1 (INSERT OR IGNORE + SELECT)
  let user: { id: number }
  try {
    user = await upsertUser(c.env.DB, wxData.openid)
  } catch (err) {
    console.error('[auth] upsertUser failed:', err)
    return fail(c, 500, 'AUTH_DB_ERROR', 'Database error during login')
  }

  // Sign JWT (HS256, 7 days = 604800s) per D-12
  const now = Math.floor(Date.now() / 1000)
  const payload = {
    sub: user.id,
    openid: wxData.openid,
    iat: now,
    exp: now + 604800,
  }
  const token = await sign(payload, c.env.JWT_SECRET)

  return ok(c, { token })
})

export default router
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| mini-program → POST /api/auth/login | wx.login code crosses from client; one-time use, treated as untrusted |
| CF Workers → api.weixin.qq.com | Workers calls out to WeChat API; response is trusted but must be validated |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Spoofing | POST /api/auth/login code parameter | mitigate | Code is forwarded directly to WeChat API for validation — WeChat validates authenticity; code is one-time use at the WeChat layer, server does not cache it |
| T-04-02 | Spoofing | JWT signing | mitigate | JWT signed with HS256 using JWT_SECRET from CF Secrets; secret never in wrangler.toml; hono/jwt sign function uses WebCrypto |
| T-04-03 | Information Disclosure | WX_APP_SECRET in fetch URL | mitigate | WX_APP_SECRET is in CF Secrets (not [vars]), injected at runtime via c.env.WX_APP_SECRET; URL is outbound HTTPS — not logged by default |
| T-04-04 | Spoofing | upsertUser race condition | accept | INSERT OR IGNORE + SELECT pattern handles concurrent logins from same openid gracefully; low risk for single-user MVP scope |
| T-04-05 | Elevation of Privilege | JWT payload sub field | mitigate | sub is set to user.id from D1 (not from request body); caller cannot self-assign a user_id |
</threat_model>
