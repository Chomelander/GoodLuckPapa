```typescript
import type { Context } from 'hono'

export function ok<T>(c: Context, data: T, status: 200 | 201 = 200) {
  return c.json({ success: true, data, error: null }, status)
}

export function fail(
  c: Context,
  status: 400 | 401 | 404 | 429 | 500,
  code: string,
  message: string
) {
  return c.json({ success: false, data: null, error: { code, message } }, status)
}
```

worker/src/lib/db.ts:
```typescript
export function toCamel(row: Record<string, unknown>): Record<string, unknown> {
  return Object.fromEntries(
    Object.entries(row).map(([k, v]) => [
      k.replace(/_([a-z])/g, (_, c) => c.toUpperCase()),
      v,
    ])
  )
}

export async function upsertUser(db: D1Database, openid: string): Promise<{ id: number }> {
  await db
    .prepare('INSERT OR IGNORE INTO users (openid) VALUES (?1)')
    .bind(openid)
    .run()
  const user = await db
    .prepare('SELECT id FROM users WHERE openid = ?1')
    .bind(openid)
    .first<{ id: number }>()
  if (!user) throw new Error('Failed to upsert user')
  return user
}
```

```typescript
import { jwt } from 'hono/jwt'
import type { MiddlewareHandler } from 'hono'
import type { Bindings, Variables } from '../types'

export const authMiddleware: MiddlewareHandler<{
  Bindings: Bindings
  Variables: Variables
}> = (c, next) => {
  // REQUIRED: explicit alg prevents CVE-2026-22817 algorithm confusion attack
  const jwtMiddleware = jwt({
    secret: c.env.JWT_SECRET,
    alg: 'HS256',
  })
  return jwtMiddleware(c, next)
}
```

```typescript
import { Hono } from 'hono'
import { cors } from 'hono/cors'
import type { Bindings, Variables } from './types'
import healthRoutes from './routes/health'
import authRoutes from './routes/auth'
import profileRoutes from './routes/profile'
import recordsRoutes from './routes/records'
import milestonesRoutes from './routes/milestones'
import diaryRoutes from './routes/diary'

const app = new Hono<{ Bindings: Bindings; Variables: Variables }>()

// CORS must come before routes
app.use('*', async (c, next) => {
  const corsMiddleware = cors({ origin: c.env.CORS_ORIGIN ?? '*' })
  return corsMiddleware(c, next)
})

app.route('/api', healthRoutes)
app.route('/api', authRoutes)
app.route('/api', profileRoutes)
app.route('/api', recordsRoutes)
app.route('/api', milestonesRoutes)
app.route('/api', diaryRoutes)

export default app
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| .dev.vars → CF Workers runtime | Secrets injected at startup; file must never be committed |
| wrangler.toml vars → public | [vars] section is NOT encrypted; secrets must never go here |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | Spoofing | middleware/auth.ts JWT verify | mitigate | Explicit `alg: 'HS256'` in jwt() call — prevents CVE-2026-22817 algorithm confusion where attacker forges HS256 token using RS256 public key |
| T-01-02 | Information Disclosure | wrangler.toml [vars] | mitigate | JWT_SECRET and WX_APP_SECRET must use .dev.vars locally and `wrangler secret put` for prod — not [vars]. Enforced by acceptance criteria grep check |
| T-01-03 | Information Disclosure | worker/.dev.vars | mitigate | Added to .gitignore; .dev.vars.example ships without real values; CI must never have .dev.vars |
| T-01-04 | Tampering | src/index.ts CORS | accept | Development uses `origin: '*'`; WeChat mini-program requests bypass browser CORS; low risk for this scaffold phase — tighten in Phase 2 deploy |
</threat_model>
